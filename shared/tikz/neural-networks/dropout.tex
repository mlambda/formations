\usepackage{formation}
\tikzset{
    cross/.pic = {
    \draw[line width=0.7mm, rotate = 45] (-#1,0) -- (#1,0);
    \draw[line width=0.7mm, rotate = 45] (0,-#1) -- (0, #1);
    }
}

  \begin{tikzpicture}

        \foreach \y in {1,...,4} {

        }
    }
    \foreach \i in {1,...,5} {

        }
    }

        \node[output] (O\y) at (12, 4 - \y) {$\sigma(\sum)$} ;
    }
    \foreach \i in {1,4,5} {
        \foreach \j in {1,...,3} {
