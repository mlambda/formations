\begin{frame}{Cost/error function}
  \begin{minipage}[l]{0.40\linewidth}
  \textbf{Mean absolute error}
    \[
      \frac{1}{n}\sum_{i=1}^n \lvert \hat{y_i} - y_i \rvert
    \]

  \textbf{Mean squared error}
    \[
      \frac{1}{n}\sum_{i=1}^n( \hat{y_i} - y_i )^2
    \]
  \end{minipage}\hfill
