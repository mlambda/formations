\begin{frame}{Probability}

  \begin{itemize}
  \item The probability of event $X$ is written $P(X)$
  \item $P(X) \in [0, 1]$
  \item $P(X)=0 \iff \text{X is impossible}$
  \item $P(X)=1 \iff \text{X is certain}$
  \item $P(\neg X) = 1 - P(X)$
  \end{itemize}
\end{frame}
